  <script>
    // Very small beginner javascript
    function updateEstimate(){
      var leds = Number(document.getElementById('leds').value);
      document.getElementById('ledCount').textContent = leds;
      // rough proxy: each LED saves ~0.05 tCO2/year
      var co2 = (leds * 0.05).toFixed(1);
      document.getElementById('co2').textContent = co2;
    }
    updateEstimate();

    function startQuiz(){
      var score = 0;
      var a1 = prompt('Q1: What gas causes most global warming? (hint: starts with C)');
      if (a1 && a1.toLowerCase().indexOf('co2') !== -1) score++;
      var a2 = prompt('Q2: Name one renewable energy source (e.g., solar, wind)');
      if (a2 && (a2.toLowerCase().indexOf('solar') !== -1 || a2.toLowerCase().indexOf('wind') !== -1)) score++;
      document.getElementById('quizResult').textContent = 'You scored ' + score + ' / 2';
    }
  </script>
